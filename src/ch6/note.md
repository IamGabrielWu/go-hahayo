### 数组的声明
var a[3]int //声明并初始化，默认为零值
a[0]=1 //赋值
b:=[3]int{1,2,3} // 声明同时初始化
c:=[2][2]int{{1,2},{3,4}} //声明并初始化多维数组
### 数组的截取
a[开始索引（包含）， 结束索引（不包含）]
a:=[...]int{1,2,3,4,5}
a[1:2] //2
a[1:3] //2,3
a[1:len(a)] //2,3,4,5
a[:3] //1,2,3
